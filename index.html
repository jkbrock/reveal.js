@Path("/")
public class GreetingEndpoint {
    @GET @Path("/greeting")
    public String greeting(@QueryParam("name") String name) {
        System.out.println(System.currentTimeMillis());
        return "Hello " + name;
    }
    void onStart(@Observes StartupEvent startup) {
        System.out.println(System.currentTimeMillis());
    }
}
